def rotate_right(val: int, shift: int) -> int:
    val &= (1 << 64) - 1  # 限制為 64 位元
    return ((val >> shift) | (val << (64 - shift))) & ((1 << 64) - 1)

def i64_to_string(val: int) -> str:
    bytes_ = []
    for i in range(8):
        byte = (val >> (8 * i)) & 0xFF
        bytes_.append(byte)
    return ''.join(chr(b) for b in bytes_)
